/**
 * Enhanced Asset Detail Page - v2
 * Features:
 * - Dynamic field rendering based on asset type
 * - Modern, attractive UI with better information hierarchy
 * - Responsive design with card-based layout
 * - Smart tab organization
 * - Status indicators and badges
 * - Better information grouping
 */

import { useState, useEffect, useMemo } from "react";
import { useParams, useNavigate } from "react-router-dom";
import { PageLoader } from "../../components/Loader/Loader.jsx";
import { ErrorNotification } from "../../components/ErrorBoundary/ErrorNotification.jsx";
import Button from "../../components/Button/Button.jsx";
import { getGroupedFields, getOrderedGroups, FIELD_GROUP_LABELS, getAssetSchema } from "../../utils/assetFieldSchema.js";
import "./AssetDetail.css";

const AssetDetail = () => {
  const { id } = useParams();
  const navigate = useNavigate();
  const [asset, setAsset] = useState(null);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState(null);
  const [activeTab, setActiveTab] = useState("specs");
  const [showActionMenu, setShowActionMenu] = useState(false);
  const [expandedGroups, setExpandedGroups] = useState({});

  useEffect(() => {
    const loadAsset = async () => {
      try {
        setLoading(true);
        setError(null);
        const itemType = localStorage.getItem('lastItemType') || 'cpu';
        const response = await fetch(`/api/v1/assets/${id}?itemType=${itemType}`, {
          headers: {
            'Authorization': `Bearer ${localStorage.getItem('accessToken')}`
          }
        });
        
        if (!response.ok) {
          throw new Error(`Failed to load asset: ${response.status}`);
        }
        
        const data = await response.json();
        setAsset(data.data || data);
        
        // Initialize all groups as expanded
        const groupedFields = getGroupedFields(data.data || data, itemType);
        const orderedGroups = getOrderedGroups(groupedFields);
        const initialExpanded = {};
        orderedGroups.forEach(group => {
          initialExpanded[group] = true;
        });
        setExpandedGroups(initialExpanded);
      } catch (err) {
        setError(err.message || "Failed to load asset details");
      } finally {
        setLoading(false);
      }
    };

    if (id) loadAsset();
  }, [id]);

  // Memoize grouped fields to avoid recalculation
  const groupedFields = useMemo(() => {
    if (!asset) return {};
    const itemType = localStorage.getItem('lastItemType') || 'cpu';
    return getGroupedFields(asset, itemType);
  }, [asset]);

  const orderedGroups = useMemo(() => {
    return getOrderedGroups(groupedFields);
  }, [groupedFields]);

  const itemType = localStorage.getItem('lastItemType') || 'cpu';
  const schema = useMemo(() => getAssetSchema(itemType), [itemType]);

  if (loading) return <PageLoader message="Loading asset details..." />;
  if (error) return <ErrorNotification error={error} onClose={() => setError(null)} />;

  if (!asset) {
    return (
      <div className="asset-detail-v2">
        <div className="detail-back">
          <Button variant="secondary" size="small" onClick={() => navigate('/assets')}>
            ← Back to Assets
          </Button>
        </div>
        <div className="empty-state">
          <div className="empty-icon">📭</div>
          <div className="empty-text">Asset not found</div>
        </div>
      </div>
    );
  }

  const handleToggleGroup = (group) => {
    setExpandedGroups(prev => ({
      ...prev,
      [group]: !prev[group]
    }));
  };

  const getStatusColor = (status) => {
    const s = String(status || '').toLowerCase();
    if (s.includes('active')) return 'status-active';
    if (s.includes('inactive')) return 'status-inactive';
    if (s.includes('repair')) return 'status-repair';
    if (s.includes('damaged')) return 'status-damaged';
    return 'status-default';
  };

  const getConditionColor = (condition) => {
    const c = String(condition || '').toLowerCase();
    if (c.includes('excellent') || c.includes('good') || c.includes('new')) return 'condition-excellent';
    if (c.includes('fair') || c.includes('average')) return 'condition-fair';
    if (c.includes('poor') || c.includes('broken')) return 'condition-poor';
    return 'condition-default';
  };

  const formatValue = (value, field) => {
    if (value === null || value === undefined || value === '') return 'N/A';
    if (typeof value === 'boolean') return value ? '✓ Yes' : '✗ No';
    if (field?.type === 'date') return new Date(value).toLocaleDateString();
    if (field?.type === 'number') return String(value);
    return String(value);
  };

  const totalFieldsWithValue = Object.values(groupedFields)
    .flat()
    .filter(f => f.hasValue).length;

  const totalFields = orderedGroups
    .reduce((sum, group) => sum + groupedFields[group]?.length || 0, 0);

  const completionPercentage = Math.round((totalFieldsWithValue / totalFields) * 100);

  return (
    <div className="asset-detail-v2">
      {/* Header Section */}
      <div className="detail-header-section">
        <div className="detail-back">
          <Button variant="secondary" size="small" onClick={() => navigate('/assets')}>
            ← Back to Assets
          </Button>
        </div>

        {/* Enhanced Premium Header */}
        <div className="enhanced-header">
          <div className="header-content">
            <div className="header-left">
              <div className="asset-type-badge">
                <span className="badge-icon">{schema.icon}</span>
                <span className="badge-label">{schema.label}</span>
              </div>
              <div className="header-title-section">
                <h1 className="asset-title">{asset.itemId || "Asset"}</h1>
                <p className="asset-subtitle">
                  {asset.manufacturer || 'Unknown'} • {asset.model || 'Unknown Model'}
                </p>
              </div>
            </div>

            <div className="header-right">
              {/* Status Indicators */}
              <div className="status-indicators">
                {asset.itemStatus && (
                  <div className={`status-indicator ${getStatusColor(asset.itemStatus)}`}>
                    <span className="indicator-dot"></span>
                    {asset.itemStatus}
                  </div>
                )}
                {asset.itemCondition && (
                  <div className={`condition-indicator ${getConditionColor(asset.itemCondition)}`}>
                    ⭐ {asset.itemCondition}
                  </div>
                )}
              </div>

              {/* Completion Meter */}
              <div className="completion-meter">
                <div className="meter-label">
                  <span className="meter-title">Data Completeness</span>
                  <span className="meter-percentage">{completionPercentage}%</span>
                </div>
                <div className="meter-bar">
                  <div className="meter-fill" style={{ width: `${completionPercentage}%` }}></div>
                </div>
              </div>
            </div>
          </div>

          {/* Quick Actions */}
          <div className="header-actions">
            <button className="action-btn secondary-action" title="Edit Serial Number">
              ✏️ Edit
            </button>
            <button className="action-btn secondary-action" title="View History">
              📜 History
            </button>
            <div className="action-menu-container">
              <button 
                className="action-btn primary-action"
                onClick={() => setShowActionMenu(!showActionMenu)}
                title="More Actions"
              >
                ⚙️ Actions
              </button>
              {showActionMenu && (
                <div className="action-dropdown">
                  <button className="dropdown-item">👤 Assign Asset</button>
                  <button className="dropdown-item">🔧 Log Repair</button>
                  <button className="dropdown-item">↔️ Transfer</button>
                  <button className="dropdown-item">📥 Export Data</button>
                  <button className="dropdown-item">🗑️ Soft Delete</button>
                </div>
              )}
            </div>
          </div>
        </div>
      </div>

      {/* Tab Navigation */}
      <div className="tab-navigation">
        <button 
          className={`tab-btn ${activeTab === 'specs' ? 'active' : ''}`}
          onClick={() => setActiveTab('specs')}
        >
          📋 Specifications
        </button>
        <button 
          className={`tab-btn ${activeTab === 'lifecycle' ? 'active' : ''}`}
          onClick={() => setActiveTab('lifecycle')}
        >
          📅 Lifecycle
        </button>
        <button 
          className={`tab-btn ${activeTab === 'warranty' ? 'active' : ''}`}
          onClick={() => setActiveTab('warranty')}
        >
          🛡️ Warranty
        </button>
        <button 
          className={`tab-btn ${activeTab === 'assignments' ? 'active' : ''}`}
          onClick={() => setActiveTab('assignments')}
        >
          👤 Assignments
        </button>
      </div>

      {/* Main Content Area */}
      <div className="main-content-area">
        {activeTab === 'specs' && (
          <div className="tab-content specs-content">
            {orderedGroups.map(groupKey => {
              const fields = groupedFields[groupKey] || [];
              const groupLabel = FIELD_GROUP_LABELS[groupKey];
              const isExpanded = expandedGroups[groupKey] !== false;
              const fieldsWithValue = fields.filter(f => f.hasValue).length;

              return (
                <div key={groupKey} className="spec-group">
                  <div 
                    className="group-header"
                    onClick={() => handleToggleGroup(groupKey)}
                    style={{ cursor: 'pointer' }}
                  >
                    <div className="group-title-section">
                      {/* <span className="group-icon">{groupLabel.icon}</span> */}
                      <span className="group-title">{groupLabel.label}</span>
                      <span className="group-count">{fieldsWithValue}/{fields.length}</span>
                    </div>
                    <span className={`expand-icon ${isExpanded ? 'expanded' : ''}`}>
                      ▼
                    </span>
                  </div>

                  {isExpanded && (
                    <div className="group-content">
                      <div className="fields-grid">
                        {fields.map(field => (
                          <div 
                            key={field.key} 
                            className={`field-item ${field.isEmpty ? 'empty-field' : 'filled-field'}`}
                          >
                            <div className="field-label">
                              <span className="field-icon">{field.icon}</span>
                              <span className="label-text">{field.label}</span>
                              {field.required && <span className="required-mark">*</span>}
                            </div>
                            <div className={`field-value ${field.isEmpty ? 'empty-value' : ''}`}>
                              {field.isEmpty ? (
                                <span className="placeholder-text">Not provided</span>
                              ) : (
                                <span className="actual-value">{formatValue(field.value, field)}</span>
                              )}
                            </div>
                          </div>
                        ))}
                      </div>
                    </div>
                  )}
                </div>
              );
            })}
          </div>
        )}

        {activeTab === 'lifecycle' && (
          <div className="tab-content lifecycle-content">
            <div className="info-card">
              <div className="card-header">📅 Asset Timeline</div>
              <div className="timeline-items">
                <div className="timeline-item">
                  <div className="timeline-marker">
                    <span className="marker-icon">✨</span>
                  </div>
                  <div className="timeline-info">
                    <div className="timeline-date">Created On</div>
                    <div className="timeline-value">
                      {asset.createdAt ? new Date(asset.createdAt).toLocaleDateString() : 'N/A'}
                    </div>
                  </div>
                </div>
                {asset.updatedAt && (
                  <div className="timeline-item">
                    <div className="timeline-marker">
                      <span className="marker-icon">✏️</span>
                    </div>
                    <div className="timeline-info">
                      <div className="timeline-date">Last Updated</div>
                      <div className="timeline-value">
                        {new Date(asset.updatedAt).toLocaleDateString()}
                      </div>
                    </div>
                  </div>
                )}
                {asset.isActive === false && asset.inactiveAt && (
                  <div className="timeline-item">
                    <div className="timeline-marker">
                      <span className="marker-icon">🔴</span>
                    </div>
                    <div className="timeline-info">
                      <div className="timeline-date">Inactivated On</div>
                      <div className="timeline-value">
                        {new Date(asset.inactiveAt).toLocaleDateString()}
                        {asset.inactiveReason && ` - ${asset.inactiveReason}`}
                      </div>
                    </div>
                  </div>
                )}
              </div>
            </div>
          </div>
        )}

        {activeTab === 'warranty' && (
          <div className="tab-content warranty-content">
            {asset.warranty ? (
              <div className="info-card">
                <div className="card-header">🛡️ Warranty Information</div>
                <div className="warranty-details">
                  <div className="warranty-item">
                    <span className="warranty-label">Status</span>
                    <span className={`warranty-value ${asset.warranty.status === 'Active' ? 'active' : 'inactive'}`}>
                      {asset.warranty.status}
                    </span>
                  </div>
                  {asset.warranty.expiryDate && (
                    <div className="warranty-item">
                      <span className="warranty-label">Expiry Date</span>
                      <span className="warranty-value">
                        {new Date(asset.warranty.expiryDate).toLocaleDateString()}
                      </span>
                    </div>
                  )}
                  {asset.warranty.provider && (
                    <div className="warranty-item">
                      <span className="warranty-label">Provider</span>
                      <span className="warranty-value">{asset.warranty.provider}</span>
                    </div>
                  )}
                </div>
              </div>
            ) : (
              <div className="info-card">
                <div className="card-header">🛡️ No Warranty Information</div>
                <p className="no-data-text">No warranty details found for this asset.</p>
              </div>
            )}
          </div>
        )}

        {activeTab === 'assignments' && (
          <div className="tab-content assignments-content">
            <div className="info-card">
              <div className="card-header">👤 Assignment Status</div>
              <div className="assignment-status">
                <div className="status-row">
                  <span className="status-key">Currently Assigned To</span>
                  <span className="status-val">{asset.itemUser || 'Not Assigned'}</span>
                </div>
                {asset.AssignDate && (
                  <div className="status-row">
                    <span className="status-key">Assignment Date</span>
                    <span className="status-val">
                      {new Date(asset.AssignDate).toLocaleDateString()}
                    </span>
                  </div>
                )}
                <div className="status-row">
                  <span className="status-key">Location</span>
                  <span className="status-val">{asset.itemIsCurrently || 'N/A'}</span>
                </div>
              </div>
            </div>
          </div>
        )}
      </div>
    </div>
  );
};

export default AssetDetail;
